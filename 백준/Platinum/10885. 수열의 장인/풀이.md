# 풀이
* 수열은 -2, -1, 0, 1, 2로만 구성되어 있습니다.
* 일단 목적을 위해선 0은 제외해야 합니다. 따라서, 0을 기준으로 수열을 나눠서 생각해야 합니다.
* 문제는 음수가 홀수이냐 짝수이냐를 구분해서 또 생각해 볼 수 있습니다.
* 하지만 그러면 음수 개수를 구하는 계산과 개수가 홀수냐 짝수냐 분기만 더 추가되므로 계산만 복잡해집니다.
* 그러므로 선택 집중(그리디)이 아닌 상태 집중 방법으로 해결해 봅시다.
* 음수가 홀수일 경우에만 첫 음수를 제외하고 모든 오른쪽 구간(n->0)을 계산하고 마지막 음수를 제외하고 모든 왼쪽 구간(0->n)을 계산 후 최댓값을 결정합니다.
## 문제 유형
* 수학, 그리디 알고리즘, 상태 자동자(state machine)
## 시간 및 공간 복잡도
* 시간복잡도: $\mathcal{O}(T × n)$
* 공간복잡도: $\mathcal{O}(n)$
## 코드
### Python
```python
import sys
def print(*args, sep=" ", end="\n"):
    sys.stdout.write(sep.join(map(str, args)) + end)
def input():
    return sys.stdin.readline().rstrip()

def main():
    MOD = 1_000_000_007
    T = int(input())
    for _ in range(T):
        n = int(input())
        A = list(map(int, input().split()))
        mx = -3; mxcnt = -1; cnt = [0,0]; minus = minustwo = 0; 
        flag = 0
        for a in A:
            mx = max(mx, a)
            if a > 0: flag = 1
            if a > 1: cnt[minus] += 1
            if a < 0: minus += 1
            if a < -1: minustwo += 1
            if minus == 2:
                flag = 1
                cnt[0] += minustwo+cnt[1]
                cnt[1] = minustwo = minus = 0
            if flag: mxcnt = max(mxcnt, cnt[0], cnt[1])
            if not a: flag = cnt[0] = cnt[1] = minus = minustwo = 0
        if mxcnt < 0: print(mx); continue
        cnt[0] = cnt[1] = minus = minustwo = 0
        for a in reversed(A):
            if a > 0: flag = 1
            if a > 1: cnt[minus] += 1
            if a < 0: minus += 1
            if a < -1: minustwo += 1
            if minus == 2:
                flag = 1
                cnt[0] += minustwo+cnt[1]
                cnt[1] = minustwo = minus = 0
            if flag: mxcnt = max(mxcnt, cnt[0], cnt[1])
            if not a: flag = cnt[0] = cnt[1] = minus = minustwo = 0
        print(pow(2, mxcnt, MOD))
if __name__ == '__main__':
    main()
```
### C++
```c++
#include<bits/stdc++.h>
using namespace std;
using ll = long long;
const ll MOD = 1'000'000'007;

ll pw(ll n, ll r){
	ll ret = 1;
	while(r){
		if(r&1) ret = ret*n%MOD;
		n = n*n%MOD;
		r>>=1;
	}
	return ret;
}

int main(){
    cin.tie(0); cout.tie(0); ios::sync_with_stdio(false);
    int T; cin >> T;
    while(T--){
        int n; cin >> n;
        vector<int> A(n);
        int mx = -3, mxcnt = -1, cnt[2] = {0, 0}, minus = 0, minustwo = 0;
        bool flag = 0;
        for(int i = 0; i < n; i++){
            cin >> A[i]; int x = A[i];
            mx = max(mx, x);
            if(x > 0) flag = 1;
            if(x > 1) cnt[minus]++;
            if(x < 0) minus++;
            if(x < -1) minustwo++;
            if(minus == 2){
                flag = 1;
                cnt[0] += minustwo+cnt[1];
                cnt[1] = minustwo = minus = 0;
            }
            if(flag) mxcnt = max({mxcnt, cnt[0], cnt[1]});
            if(!x) flag = cnt[0] = cnt[1] = minus = minustwo = 0;
        }
        if(mxcnt < 0){
            cout << mx << '\n'; continue;
        }
        cnt[0] = cnt[1] = minus = minustwo = 0;
        for(int i = 0; i < n; i++){
            int x = A[n-i-1];
            if(x > 0) flag = 1;
            if(x > 1) cnt[minus]++;
            if(x < 0) minus++;
            if(x < -1) minustwo++;
            if(minus == 2){
                flag = 1;
                cnt[0] += minustwo+cnt[1];
                cnt[1] = minustwo = minus = 0;
            }
            if(flag) mxcnt = max({mxcnt, cnt[0], cnt[1]});
            if(!x) flag = cnt[0] = cnt[1] = minus = minustwo = 0;
        }
        cout << pw(2, mxcnt) << '\n';
    }
    return 0;
}
```
